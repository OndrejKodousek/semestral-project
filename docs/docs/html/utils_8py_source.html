13 def shorten_string(string, max_length=60):
14  """
15  @brief Shortens a string to a specified maximum length.
16 
17  If the string is shorter than the maximum length, it will be padded with spaces.
18  If longer, it will be truncated with ellipsis in the middle.
19 
20  @param string The input string to be shortened.
21  @param max_length Maximum length of the output string (default: 60).
22  @return The shortened or padded string.
23  """
24  if len(string) <= max_length:
25  extra_space = max_length - len(string)
26  appended_spaces = extra_space * " "
27  return string + appended_spaces
28  part_length = (max_length - 3) // 2
29  return string[:part_length] + "..." + string[-part_length:]
