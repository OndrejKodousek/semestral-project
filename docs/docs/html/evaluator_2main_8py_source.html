16 def load_data(data_dir, max_day=12, min_articles_per_source=1, max_include_day=3):

20  # Initialize data structures

32 
33  for file_path in file_paths:

35  with open(file_path, "r", encoding="utf-8") as f:
36  data = json.load(f)
37  except (json.JSONDecodeError, IOError) as e:
38  print(f"Warning: Skipping {os.path.basename(file_path)}: {str(e)}")
39  continue
40 
41  model = data.get("model")
42  if model == "gemini-1.5-flash-8b":
43  continue # Skip this model
44 
45  ticker = os.path.basename(file_path).split("_")[0]
46  predictions_by_article = data.get("predictionsByArticle", [])
47 
48  if not model or not predictions_by_article:
49  continue
50 
51  for article_data in predictions_by_article:
52  source = article_data.get("source")
53  predictions = article_data.get("predictions", {})
54 
55  if (
56  not source
57  or not predictions
58  or len(predictions) < min_articles_per_source
59  ):
60  continue
61 
62  for date, values in predictions.items():
63  prediction = values.get("prediction")
64  real = values.get("real")
65  day = values.get("predictionDay")
66 
67  if (
68  day is None
69  or day < 1
70  or day > max_day
71  or real is None
72  or prediction is None
73  or real == 0
74  or day > max_include_day
75  ):
76  continue
77 
78  percentage_error = abs(prediction - real) / abs(real)
79 
80  # Track metrics by day (all models/sources)
81  day_metrics[day]["errors"].append(percentage_error)
82  day_metrics[day]["count"] += 1
83 
84  # Track metrics by model and day
85  model_day_metrics[model][day].append(percentage_error)
86  model_counts[model][day] += 1
87  model_overall[model].append(percentage_error)
88 
89  # Track metrics by source and day
90  source_day_metrics[source][day].append(percentage_error)
91  source_counts[source][day] += 1
92  source_overall[source].append(percentage_error)
93 
94  # Track metrics by ticker
95  ticker_metrics[ticker][day].append(percentage_error)
96 
97  return {
98  "day_metrics": day_metrics,
99  "model_day_metrics": model_day_metrics,
100  "source_day_metrics": source_day_metrics,
101  "model_counts": model_counts,
102  "source_counts": source_counts,

109 def plot_accuracy_over_time(data, min_datapoints=10, output_dir="graphs"):

135  if day_counts[i] > 0:
136  ax.text(
137  day,

140  ha="center",
141  va="bottom",
142  fontsize=8,
143  )
144 
145  ax.set_xlabel("Prediction Day")
146  ax.set_ylabel("Mean Absolute Percentage Error (MAPE) %")
147  ax.set_title("Prediction Accuracy Degradation Over Time")

149  ax.xaxis.set_major_locator(MaxNLocator(integer=True))

151  plt.tight_layout()
