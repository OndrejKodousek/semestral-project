149  * @param values - Array of numeric values
150  * @returns Calculated volatility
151  */
152 const calculateVolatility = (values: number[]): number => {
153  const avg = values.reduce((a, b) => a + b, 0) / values.length;
154  const variance =
155  values.reduce((a, b) => a + Math.pow(b - avg, 2), 0) / values.length;
156  return Math.sqrt(variance);
